    def tree(d,r):
    if d&lt;7 or r&lt;10:
        return
    else:
        turtle.pensize(d/10)
        turtle.forward(d)
        turtle.left(r)
        tree(d*0.7,r)
        turtle.pensize(d/80)
        turtle.right(2*r)
        tree(d*0.7,r)
        turtle.pensize(d/95)
        turtle.left(r)
        turtle.backward(d)
        turtle.fillcolor("pink")
        turtle.begin_fill()
        star(5,10)
        turtle.end_fill()
        
        turtle.fillcolor("green")
        turtle.begin_fill()
        star(7,10)
        turtle.end_fill()
    turtle.tracer(0)
    turtle.left(90)
    turtle.pencolor("brown")
    tree(100,30)
    turtle.update()
    turtle.mainloop()
